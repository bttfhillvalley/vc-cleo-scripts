{$INCLUDE Variables.txt}

:DigitalSpeedoReset
3F10: set_car vehicle component "digitalspeedodigit20bttf3" visible 0
for var3 = 10 to 29
    3F11: set_car vehicle component "digitalspeedodigit" index var3 visible 0
end
return

:DigitalSpeedo
if
    Car.Wrecked(vehicle)
then
    return
end
02E3: var1 = car vehicle speed
var1 *= 1.835
008C: var1 = float_to_integer var1
// Limit to 88 mph
if
    var1 > 88
then
    var1 = 88
end
var2 = var1
var2 /= 10
var2 *= 10
var1 -= var2
var2 /= 10

// Show Digits
cleo_call @DigitOn 4 vehicle "digitalspeedodigit1" var1 9
if
    not var2 == 0
then
    cleo_call @DigitOn 4 vehicle "digitalspeedodigit2" var2 9
else
    cleo_call @DigitOn 4 vehicle "digitalspeedodigit2" -1 9
end
return
