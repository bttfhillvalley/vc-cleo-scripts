// Already using 1@-6@ in other scripts, will need to create a second one in case we need more animations.
// 7@ - Left window
// 8@ - Right window
// 9@ - 0: Windows up, 1: Left down, 2: Right down, 3: Both
// 10@ - Blinker state
// 11@ Reserved for acceleration key status
// 12@ Reserved for Engine state

gosub @CarClockReset
gosub @DigitalSpeedoReset
gosub @ShifterReset
while true
wait 10
    if
        Car.Wrecked(vehicle)
    then
        terminate_this_custom_script
    end
    gosub @CarClock
    gosub @Compass
    gosub @DigitalSpeedo
    gosub @DoorCheck
    gosub @Horn
    gosub @Ignition
    gosub @ParkingBrake
    gosub @Shifter
    gosub @Windows
end
