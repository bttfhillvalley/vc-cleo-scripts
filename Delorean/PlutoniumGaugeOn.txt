{$CLEO .s}
{$INCLUDE ../Common/CleoConstants.txt}
0000:
int vehicle
int vehicle_flags
int flag
float needle_angle
float main_angle
float secondary_angle
int tcon_var

while timera < 2000
    wait 10
    0AB4: tcon_var = var TCON
    if
        tcon_var == 0
    then
        terminate_this_custom_script
    end
end
3F10: set_car vehicle component "gloveboxgaugeslights" visible 1
0B10: flag = vehicle_flags AND FUEL_BIT
if
    flag == 0
then
    terminate_this_custom_script
end
3F84: play_sound "plut_gauge.wav" loop 0

// Eventually this will be part of Ilufir's animation handler so we don't have to code it here.
for needle_angle = 0.0 to 90.0 step 2.0
    wait 10
    0AB4: tcon_var = var TCON
    if
        tcon_var == 0
    then
        terminate_this_custom_script
    end
    main_angle = needle_angle
    main_angle /= 2.0
    if
        main_angle < 10.0
    then
        main_angle = 10.0
    end
    secondary_angle = needle_angle
    secondary_angle /= 3.0
    if
        secondary_angle >= 22.5
    then
        secondary_angle = 22.5
    end
    3F16: rotate_car_part "pchamberneedle" angle 0.0 main_angle 0.0 car vehicle
    3F16: rotate_car_part "ppowerneedle" angle 0.0 secondary_angle 0.0 car vehicle
    3F16: rotate_car_part "primaryneedle" angle 0.0 secondary_angle 0.0 car vehicle
end
terminate_this_custom_script
