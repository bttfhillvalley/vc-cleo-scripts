{$CLEO .s}
{$INCLUDE ../Common/CleoConstants.txt}
0000:
int vehicle
int chaser
int wheel_state
int engine
int 16@
while true
    wait 0
    if
        Car.Wrecked(vehicle)
    then
        terminate_this_custom_script
    end
    // Reset lights
    for chaser = 1 to 5
        3F11: set_car vehicle component "chaserlights" index chaser visible 0
    end
    3F02: engine =  car vehicle engine_state
    if
        3F40:   is_car vehicle component "underbodybttf2" visible
    then
        3F27: get_car vehicle wheel_status_to wheel_state
        if
            wheel_state == 2  // Wheels removed when hovering
        then
            cleo_call @FadeComponent 4 vehicle "bottomlights" 255 15
            if
                timera >= 1250
            then
                timera = 0
            end
            chaser = timera
            chaser /= 250
            chaser += 1
            3F11: set_car vehicle component "chaserlights" index chaser visible 1
        else
            cleo_call @FadeComponent 4 vehicle "bottomlights" 0 15
        end
    end
end

{$INCLUDE ../Common/FadeComponent.txt}
