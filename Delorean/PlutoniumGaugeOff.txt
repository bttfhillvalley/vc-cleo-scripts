{$CLEO .s}
{$INCLUDE ../Common/CleoConstants.txt}
0000:
int vehicle
int vehicle_flags
int flag
float needle_angle
float main_angle
float secondary_angle
int tcon_var
0B10: flag = vehicle_flags AND FUEL_BIT
if
    flag == 0
then
    terminate_this_custom_script
end
for needle_angle = 90.0 downto 0.0 step 2.0
    wait 10
    0AB4: tcon_var = var TCON
    if
        tcon_var == 1
    then
        terminate_this_custom_script
    end
    0087: main_angle = needle_angle
    main_angle /= 2.0
    if
        main_angle < 10.0
    then
        main_angle = 10.0
    end
    0087: secondary_angle = needle_angle
    secondary_angle /= 3.0
    if
        secondary_angle >= 22.5
    then
        secondary_angle = 22.5
    end
    3F16: rotate_car_part "pchamberneedle" angle 0.0 main_angle 0.0 car 0@
    3F16: rotate_car_part "ppowerneedle" angle 0.0 secondary_angle 0.0 car 0@
    3F16: rotate_car_part "primaryneedle" angle 0.0 secondary_angle 0.0 car 0@
end
terminate_this_custom_script
