{$CLEO .cs}
{$INCLUDE Common/CleoConstants.txt}
const
    KEY_TOGGLEMODE = 77
end
0000:
thread 'MODE'
int mode_var = 0
int travel_var
int cutscene_var
int camera_var
while true
    wait 10
    0AB4: travel_var = var TRAVEL
    0AB4: cutscene_var = var CUTSCENE
    0AB4: camera_var = var CAMERA_CONTROL
    if and
        travel_var == 0 // integer values
        cutscene_var == 0 // integer values
        camera_var == 1
        is_key_pressed KEY_TOGGLEMODE // M
    then
        mode_var += 1 // integer values
        if
            mode_var == 1 // integer values
        then
            03E5: text_box 'MODE1'  // Mode: Instant Time Travel Sequence
        else
            mode_var = 0 // integer values
            03E5: text_box 'MODE2'  // Mode: Cutscene Time Travel Sequence
        end
        0AB3: var MODE = mode_var
        wait 500
    end
end
